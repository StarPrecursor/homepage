import concurrent.futures
import logging
import pathlib
from typing import Tuple, Union

import matplotlib
import matplotlib.pyplot as plt
import numpy as np

from smdt_rdv.common import smdt_info
from smdt_rdv.common.smdt_const import *

logger = logging.getLogger("smdt_rdv")

matplotlib.use("Agg")
matplotlib.rc("figure", max_open_warning=0)

tdc_time_resolution = (
    25 / 32
)  # Note: coarse time resolution 25 ns, 5 bits fine time for AMT (2^5 = 32)
# HPTDC have 2 more bits for time record, divided by 4 to be unified with AMT during the database dumping
# so same resolution 25/32 can be used for both AMT and HPTDC
SPECTRUM_RANGE = 2048  # add protection for potential overflow? although this range should cover according to current data


def init_spectrums_dict() -> dict:
    """Initializes the spectrum dictionary for all mezzs and channels

    Returns:
        dict: spectrum dictionary
    """
    smdt_spectrums = {}
    for mezz in range(MAXNUMBERMEZZANINE):
        smdt_spectrums[mezz] = {}
        for channel in range(MAXNUMBERCHANNEL):
            smdt_spectrums[mezz][channel] = init_spectrums_channel()
    return smdt_spectrums


def init_spectrums_channel() -> dict:
    """Initialize the spectrum objects in one channel

    Returns:
        dict: dictionary of spectrums in one channel
    """
    channel_spectrums = {}
    # all event hits
    channel_spectrums["tdc_lead"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    channel_spectrums["tdc_trail"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    channel_spectrums["adc"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    # hits on track
    channel_spectrums["tdc_lead_track"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    channel_spectrums["tdc_trail_track"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    channel_spectrums["tdc_t0_track"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    channel_spectrums["adc_track"] = np.zeros(SPECTRUM_RANGE, dtype=np.uint32)
    return channel_spectrums


def is_mezz_exist(chamber_name: str, mezz: int) -> bool:
    """Checks where a mezzanine card location of a chamber exists in real world

    Args:
        chamber_name (str): chamber name
        mezz (int): mezzanine card id

    Returns:
        bool: [description]
    """
    mezz_map = smdt_info.smdt_mezz_map_dict
    if chamber_name in mezz_map:
        if mezz in getattr(smdt_info, mezz_map[chamber_name]):
            return True
        else:
            return False
    else:
        logger.error(
            f"Can't find mezzanine card map for chamber {chamber_name}, please add information to smdt_mezz_map in smdt_rdv.common.smdt_info!"
        )
        return False


def merge_spectrums(current_spectrums: dict, append_spectrums: dict):
    """Merges append spectrums to current spectrums
    """
    current_spectrums["tdc_lead"] += append_spectrums["tdc_lead"]
    current_spectrums["tdc_trail"] += append_spectrums["tdc_trail"]
    current_spectrums["adc"] += append_spectrums["adc"]
    current_spectrums["tdc_lead_track"] += append_spectrums["tdc_lead_track"]
    current_spectrums["tdc_trail_track"] += append_spectrums["tdc_trail_track"]
    current_spectrums["tdc_t0_track"] += append_spectrums["tdc_t0_track"]
    current_spectrums["adc_track"] += append_spectrums["adc_track"]


def merge_spectrums_list(spectrums_list: list) -> dict:
    """Merges a list of smdt_spectrums defined in init_spectrums_dict

    Args:
        spectrums_list (list): list of smdt_spectrums

    Returns:
        dict: merged smdt_spectrums
    """
    out_spectrums = init_spectrums_dict()
    for spectrums in spectrums_list:
        for mezz in range(MAXNUMBERMEZZANINE):
            for channel in range(MAXNUMBERCHANNEL):
                # current output spectrums
                current_spectrums = out_spectrums[mezz][channel]
                # new spectrums contents
                new_spectrums = spectrums[mezz][channel]
                # update
                merge_spectrums(current_spectrums, new_spectrums)
    return out_spectrums


def merge_channels_to_mezz(spectrums_dict: dict, mezz: int) -> dict:
    """Merges channels spectrums in specified mezz card

    Args:
        spectrums_dict (dict): spectrum dictionary
        mezz (int): mezzanine card id

    Returns:
        dict: merged smdt_spectrums
    """
    out_spectrums = init_spectrums_channel()
    channel_spectrums = spectrums_dict[mezz]
    for channel in range(MAXNUMBERCHANNEL):
        channel_spectrum = channel_spectrums[channel]
        # update
        merge_spectrums(out_spectrums, channel_spectrum)
    return out_spectrums


def plot_single_spectrum(
    input_spectrum: np.ndarray,
    title: str,
    save_path: str,
    unit: str = "digit",
    plot_bins: int = 200,
    plot_range: tuple = None,
) -> None:
    """Make one spectrum plot

    Args:
        input_spectrum (np.ndarray): length is spectrum range and bin contents are bin values
        title (str): title show in plot
        save_path (str): path to save plot
        unit (str, optional): "digit" to show time in digit, "ns" to show in nano seconds. Defaults to "digit".
        plot_bins (int, optional): number of bins in the spectrum
        plot_range (tuple, optional): plot range. Defaults to None.

    """
    # process data
    non_zero_ids = np.argwhere(input_spectrum > 0)
    bin_centers = np.array(range(len(input_spectrum)))[non_zero_ids]
    bin_weights = input_spectrum[non_zero_ids]
    # plot
    fig, ax = plt.subplots(figsize=(8, 6))
    ax.set_title(title)
    ax.set_xlabel("time (digit)")
    ax.set_ylabel("counts")
    if unit == "ns":
        bin_centers *= tdc_time_resolution
        ax.set_xlabel("time (ns)")
    elif unit == "digit":
        pass
    else:
        logger.warn(f"Unknown time unit {unit}, use default: digit")
    ax.hist(bin_centers, bins=plot_bins, range=plot_range, weights=bin_weights)
    fig.savefig(save_path)
    plt.close()


def plot_spectrums(
    spectrums_dict: dict,
    chamber_name: str,
    save_dir=".",
    unit: str = "digit",
    plot_bins_tdc: int = 200,
    plot_range_tdc: Tuple[float, float] = None,
    plot_bins_adc: int = 200,
    plot_range_adc: Tuple[float, float] = None,
    max_workers: int = 1,
) -> None:
    """Makes TDC/ADC spectrums plots

    Args:
        spectrums_dict (dict): spectrum dictionary
        chamber_name (str): chamber name
        save_dir (str, optional): directory to save the plots. Defaults to ".".
        unit (str, optional): "digit" to show time in digit, "ns" to show in nano seconds. Defaults to "digit".
        plot_bins (int, optional): number of bins in the spectrum
        plot_range_tdc (Tuple, optional): range for TDC plots. Defaults to None.
        plot_range_adc (Tuple, optional): range for ADC plots. Defaults to None.
    """
    logger.info("Ploting spectrums...")
    with concurrent.futures.ProcessPoolExecutor(max_workers=max_workers) as executor:
        for mezz in range(MAXNUMBERMEZZANINE):
            if not is_mezz_exist(chamber_name, mezz):
                continue
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/tdc_lead").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/tdc_trail").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/adc").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/tdc_lead_track").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/tdc_trail_track").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/tdc_t0_track").mkdir(
                parents=True, exist_ok=True
            )
            pathlib.Path(f"{save_dir}/mezz_{mezz:02d}/adc_track").mkdir(
                parents=True, exist_ok=True
            )

            # plot spectrums in each mezz
            mezz_spectrums = merge_channels_to_mezz(spectrums_dict, mezz)
            # plot TDC leading edge spectrums
            tdc_lead_spectrum = mezz_spectrums["tdc_lead"]
            title = f"mezz_{mezz:02d}_tdc_lead_spectrums"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                tdc_lead_spectrum,
                title,
                save_path,
                unit,
                plot_bins_tdc,
                plot_range_tdc,
            )
            # plot TDC trailing edge spectrums
            tdc_trail_spectrum = mezz_spectrums["tdc_trail"]
            title = f"mezz_{mezz:02d}_tdc_trail_spectrums"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                tdc_trail_spectrum,
                title,
                save_path,
                unit,
                plot_bins_tdc,
                plot_range_tdc,
            )
            # plot ADC spectrums
            adc_spectrum = mezz_spectrums["adc"]
            title = f"mezz_{mezz:02d}_adc_spectrums"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                adc_spectrum,
                title,
                save_path,
                unit,
                plot_bins_adc,
                plot_range_adc,
            )
            # plot TDC leading edge spectrums on track
            tdc_lead_spectrum = mezz_spectrums["tdc_lead_track"]
            title = f"mezz_{mezz:02d}_tdc_lead_spectrums_on_track"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                tdc_lead_spectrum,
                title,
                save_path,
                unit,
                plot_bins_tdc,
                plot_range_tdc,
            )
            # plot TDC trailing edge spectrums on track
            tdc_trail_spectrum = mezz_spectrums["tdc_trail_track"]
            title = f"mezz_{mezz:02d}_tdc_trail_spectrums_on_track"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                tdc_trail_spectrum,
                title,
                save_path,
                unit,
                plot_bins_tdc,
                plot_range_tdc,
            )
            # plot TDC - T0 spectrums on track
            tdc_trail_spectrum = mezz_spectrums["tdc_t0_track"]
            title = f"mezz_{mezz:02d}_tdc_t0_spectrums_on_track"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                tdc_trail_spectrum,
                title,
                save_path,
                unit,
                plot_bins_tdc,
                plot_range_tdc,
            )
            # plot ADC spectrums on track
            adc_spectrum = mezz_spectrums["adc_track"]
            title = f"mezz_{mezz:02d}_adc_spectrums_on_track"
            save_path = f"{save_dir}/mezz_{mezz:02d}/{title}.png"
            executor.submit(
                plot_single_spectrum,
                adc_spectrum,
                title,
                save_path,
                unit,
                plot_bins_adc,
                plot_range_adc,
            )

            # plot spectrums in each channel
            for channel in range(MAXNUMBERCHANNEL):
                current_spectrums = spectrums_dict[mezz][channel]
                # plot TDC leading edge spectrums
                tdc_lead_spectrum = current_spectrums["tdc_lead"]
                title = f"mezz_{mezz:02d}_ch_{channel}_tdc_lead_spectrums"
                save_path = f"{save_dir}/mezz_{mezz:02d}/tdc_lead/{title}.png"
                executor.submit(

                    title,
                    save_path,
                    unit,

                    plot_range_tdc,
                )
                # plot TDC trailing edge spectrums
                tdc_trail_spectrum = current_spectrums["tdc_trail"]
                title = f"mezz_{mezz:02d}_ch_{channel}_tdc_trail_spectrums"
                save_path = f"{save_dir}/mezz_{mezz:02d}/tdc_trail/{title}.png"
                executor.submit(
                    plot_single_spectrum,
                    tdc_trail_spectrum,
                    title,
                    save_path,
                    unit,
                    plot_bins_tdc,
                    plot_range_tdc,
                )
                # plot ADC spectrums
                adc_spectrum = current_spectrums["adc"]
                title = f"mezz_{mezz:02d}_ch_{channel}_adc_spectrums"
                save_path = f"{save_dir}/mezz_{mezz:02d}/adc/{title}.png"
