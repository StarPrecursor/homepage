import logging
from typing import Tuple, Union

import matplotlib
import matplotlib.pyplot as plt

from smdt_rdv.common.mapping import *
from smdt_rdv.common.smdt_const import *
from smdt_rdv.common.smdt_info import *

matplotlib.use("Agg")
matplotlib.rc("figure", max_open_warning=0)

logger = logging.getLogger("smdt_rdv")


def close_figures(plt_close_arg=None) -> None:
    """Wrapper of plt.close()

    Args:
        plt_close_arg (optional): argument passed to plt.close(). Defaults to None.
    """
    plt.close(plt_close_arg)


def create_smdt_tubes_axes(title: str = "sMDT hits display") -> (plt.figure, plt.axes):
    """Creates figure and axes objects for track/hits display

    Args:
        title (str, optional): title of the plot name. Defaults to "sMDT hits display")->(plt.figure.

    Returns:
        plt.figure: matplotlib figure object for track/hits display
        plt.axes: matplotlib axes object for track/hits display
    """
    fig, ax = plt.subplots(figsize=(16, 6))
    ax.set_title(title)
    ax.set_xlabel("(mm)")
    ax.set_ylabel("(mm)")
    ax.set_aspect("equal", "box")
    return fig, ax


def plot_smdt_tubes(ax: plt.axes, chamber_name: str) -> plt.axes:
    """Plots 2D sMDT tubes' geometry

    Args:
        ax (plt.axes): matplotlib axes object for plotting
        chamber_name (str): chamber name

    Returns:
        plt.axes: axes object with tubes' geometry plotted 
    """
    # check chamber information for plotting
    chamber_category = chamber_name[-3:]
    if chamber_category in ["A02", "A04", "A06"]:
        csm1_map = csm1_map_a2
        csm2_map = csm2_map_a2
        mezz_map = mezz_a_map
        is_a_side = True
    elif chamber_category in ["C02", "C04", "C06"]:
        csm1_map = csm1_map_c2
        csm2_map = csm2_map_c2
        mezz_map = mezz_c_map
        is_a_side = False
    elif chamber_category in ["A08", "A10", "C12", "A14", "A16"]:
        csm1_map = csm1_map_a8
        csm2_map = csm2_map_a8
        mezz_map = mezz_a_map
        is_a_side = True
    elif chamber_category in ["C08", "C10", "A12", "C14", "C16"]:
        csm1_map = csm1_map_c8
        csm2_map = csm2_map_c8
        mezz_map = mezz_c_map
        is_a_side = False
    else:
        logger.error(f"Unsupported chamber: {chamber_name}")

    if "BIX7" in chamber_name:
        csm_maps = [csm1_map]
        x_length = SMDT_TUBE_RADIUS + SMDT_TUBE_DISTANCE * 54
        y_length = (
            2 * SMDT_TUBE_RADIUS
            + 2 * (NUM_MULTILAYER - 1) * SMDT_LAYER_DISTANCE
            + SMDT_SPACER_DISTANCE
        )
    elif "BIY7" in chamber_name:
        csm_maps = [csm2_map]
        x_length = SMDT_TUBE_RADIUS + SMDT_TUBE_DISTANCE * 54
        y_length = (
            2 * SMDT_TUBE_RADIUS
            + 2 * (NUM_MULTILAYER - 1) * SMDT_LAYER_DISTANCE
            + SMDT_SPACER_DISTANCE
        )
    elif "BIS7" in chamber_name:
        if is_a_side:
            csm_maps = [csm2_map, csm1_map]
        else:
            csm_maps = [csm1_map, csm2_map]
        x_length = SMDT_TUBE_RADIUS + SMDT_TUBE_DISTANCE * 54 * 2
        y_length = (
            2 * SMDT_TUBE_RADIUS
            + 2 * (NUM_MULTILAYER - 1) * SMDT_LAYER_DISTANCE
            + SMDT_SPACER_DISTANCE
        )
    else:
        logger.error(f"Unsupported chamber: {chamber_name}")

    # make plots
    if is_a_side:
        center_x_shift = 1
    else:
        center_x_shift = 0
    ax.set_aspect("equal", "box")
    ax.axis([0, x_length, 0, y_length])
    for csm_num, csm_map in enumerate(csm_maps):
        for mezz_layer, mezz_column in csm_map.values():
            for tube_layer, tube_column in mezz_map.values():
                layer = mezz_layer * NUM_MEZZ_LAYER + tube_layer
                column = (
                    csm_num * NUM_CSM_CHANNEL / 2 + mezz_column
                ) * NUM_MEZZ_COLUMN + tube_column
                center_x = (
                    SMDT_TUBE_RADIUS
                    + column * SMDT_TUBE_DISTANCE
                    + ((layer + center_x_shift) % 2) * SMDT_TUBE_DISTANCE / 2.0
                )
                if layer < NUM_MULTILAYER:
                    center_y = SMDT_TUBE_RADIUS + layer * SMDT_LAYER_DISTANCE
                else:
                    center_y = (
                        SMDT_TUBE_RADIUS
                        + (layer - 1) * SMDT_LAYER_DISTANCE
                        + SMDT_SPACER_DISTANCE
                    )
                logger.debug(f"layer {layer}, colum {column}")
                logger.debug(f"x {center_x}, y (center_y)")
                if (column // NUM_MEZZ_COLUMN) % 2 == 0:
                    fill_color = "silver"
                else:
                    fill_color = "ivory"
                ax.add_patch(
                    plt.Circle(
                        (center_x, center_y),
                        SMDT_TUBE_RADIUS,
                        facecolor=fill_color,
                        edgecolor="royalblue",
                        alpha=0.5,
                    )
                )

    return ax


def plot_hit_layer_column(
    ax: plt.axes,
    chamber_name: str,
    layer: int,
    column: int,
    radius: float = SMDT_TUBE_RADIUS,
) -> plt.axes:
    """Plots single hit

    Args:
        ax (plt.axes): matplotlib axes object for plotting
        chamber_name (str): chamber name
        layer (int): layer of the tubes, 0~8 from bottom to top
        column (int): column of the tubes, 0~54 from left to right
        radius (float, optional): radius of the hit (via R-T function). Defaults to SMDT_TUBE_RADIUS.

    Returns:
        plt.axes: axes object with single hit plotted 
    """
    center_x, center_y = tube_layer_column_to_x_y(chamber_name, layer, column)
    ax.add_patch(
        plt.Circle(
            (center_x, center_y), radius, facecolor="gold", edgecolor="none", alpha=0.8,
        )
    )
    return ax


def plot_hit_x_y(
    ax: plt.axes,
    center_x: float,
    center_y: float,
    radius: float = SMDT_TUBE_RADIUS,
    fill_color: str = "gold",
    mark_hits: bool = True,
    mark_color: str = "black",
) -> plt.axes:
    """Plots single hit

    Args:
        ax (plt.axes): matplotlib axes object for plotting
        center_x (float): x coordinate of tube center where hit happened
        center_y (float): y coordinate of tube center where hit happened
        radius (float, optional): radius of the hit (via R-T function). Defaults to SMDT_TUBE_RADIUS.

    Returns:
        plt.axes: axes object with single hit plotted 
    """
    ax.add_patch(
        plt.Circle(
            (center_x, center_y),
            radius,
            facecolor=fill_color,
            edgecolor="none",
            alpha=0.8,
        )
    )
    if mark_hits:
        ax.add_patch(
            plt.Circle(
                (center_x, center_y),
                SMDT_TUBE_RADIUS,
                fill=False,
                edgecolor=mark_color,
                alpha=0.8,
            )
        )
    return ax


def save_hits(
    valid_hits: list, chamber_name: str, plot_title: str, save_path: str
) -> None:
    """Saves the plots of selected hits

    Args:
        valid_hits (list): tdc leading edges information
        chamber_name (str): chamber name
        plot_title (str): title of the saved plot
        save_path (str): path to save the plot, this also specify the save format
    """
    fig, ax = create_smdt_tubes_axes(title=plot_title)
    plot_smdt_tubes(ax, chamber_name)
    for (mezz, channel, _, _, _,) in valid_hits:
        # plot hit based on leading TDC
        hit_layer, hit_column = mezz_to_tube_layer_column(chamber_name, mezz, channel)
        plot_hit_layer_column(ax, chamber_name, hit_layer, hit_column)
    fig.savefig(save_path)
    close_figures()
