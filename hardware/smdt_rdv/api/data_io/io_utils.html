import logging
from functools import partial
from io import DEFAULT_BUFFER_SIZE

import numpy as np

from smdt_rdv.common.smdt_const import *
from smdt_rdv.common.smdt_info import mrod_td_type

logger = logging.getLogger("smdt_rdv")


def check_hits_quantity(tdc_data_array: np.ndarray, tdc_type: int):
    """ Identify and process TDC data word

    Args:
        tdc_data_array (int): 32bit TDC data word
        tdc_type (int): tdc_type 0 (for AMT) or 1 (for HPTDC)

    Returns:
        num_hits (num_hits): number of hits counted

    """
    got_twc = False
    num_hits = 0
    for i, word_td in enumerate(tdc_data_array):
        word_header = (word_td >> MAINIDBIT0LOCATION) & MAINIDBITS
        word_sub_header = (word_td >> SUBIDBIT0LOCATION) & SUBIDBITS
        if (tdc_type == 0) and (word_header == TDC_PAIR):
            num_hits += 1
        elif (tdc_type == 0) and (word_header == TDC_EDGE):
            edge_type = (word_td >> 18) & 0x1
            if edge_type == 1:
                num_hits += 1  # Leading Edge
        elif (tdc_type == 1) and (word_header == HPTDC_LEADING):
            num_hits += 1
        elif (word_header == MROD_BOBLOCK) and (
            word_sub_header == 0xA
        ):  # Got a MROD TWC (Trailer Word Count) word
            got_twc = True
            if i != len(tdc_data_array) - 1:
                logger.warn(
                    "MROD Event Size smaller than expected from LWC (Link Word Count) word"
                )
    if not got_twc:
        logger.warn(
            "MROD Event Size larger than expected from LWC (Link Word Count) word"
        )
    return num_hits


def file_byte_iterator(path):
    """Returns an iterator over the file that lazily loads the file"""
    with open(path, mode="rb") as file:
        reader = partial(file.read1, DEFAULT_BUFFER_SIZE)
        file_iterator = iter(reader, bytes())
        for chunk in file_iterator:
            yield from chunk


def get_mrod_header(data_word: int):
    """Returns main and sub ID of data word"""
    header = (data_word >> MAINIDBIT0LOCATION) & MAINIDBITS
    sub_header = (data_word >> SUBIDBIT0LOCATION) & SUBIDBITS
    return header, sub_header


def get_mrod_td_type(td_data_word: int, tdc_type: int):
    """Decides the type a TDC data word

    Args:
        td_data_word (int): TDC data word
        tdc_type (int): tdc_type 0 (for tdc_id between 0~15) or 1 (between 16~17)
    """
    td_header, _ = get_mrod_header(td_data_word)
    search_key = (td_header, tdc_type)
    if search_key in mrod_td_type:
        return mrod_td_type[(td_header, tdc_type)]
    else:
        return "unknown"
