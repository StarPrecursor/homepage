import copy
import logging
import math
import warnings
from typing import Iterable, Tuple

import numpy as np
import scipy.optimize
import scipy.stats

from smdt_rdv.analyze.rt_function import rt_err_function, rt_function
from smdt_rdv.common.smdt_const import *

logger = logging.getLogger("smdt_rdv")
warnings.filterwarnings("ignore")


def fit_track(
    points_xy: Iterable[Tuple[float, float]],
    tdc_times: list,
    track_hits_index: list = [],
    track_hits: list = [],
    noise_hits: list = [],
    fit_method: str = "SLSQP",
    negative_repel_factor: float = 5,
    off_track_hit_err: float = 7,
    k_angle_bound_range: float = 0.1963,  # pi/16
    b_bound_range: float = 200,
    max_iter: int = 50,
    iter_tolerance: int = 1,
) -> Tuple[float, float, float, float]:
    """Fits the cosmic track with given hits

    Args:
        points_xy (Iterable[Tuple[float, float]]): points locations described by coordinates
        tdc_times (list): tdc times (unit: ns)
        track_hits_index (list, optional): where the track hits ids written to. Defaults to [].
        track_hits (list, optional): where the track hits (x, y, radius) written to. Defaults to [].
        noise_hits (list, optional): where the noise hits (x, y, radius) written to. Defaults to [].
        fit_method (str, optional): fitting algorithm. Defaults to "SLSQP".
        negative_repel_factor (float, optional): the factor to determine how far you want to refuse the events which has
            smaller point to track distance than hit radius . Defaults to 5.
        off_track_hit_err (float, optional): criteria to decide a hit is off-track. Defaults to 7.
        k_angle_bound_range (float, optional): set bound of fitting for track line slope (range: -pi/2 = pi/2)
        b_bound_range (float, optional): set bound of fitting for track line intersection
        max_iter (int, optional): maximum iterations of optimization
        iter_tolerance (int, optional): how many times to tolerace worse optimization

    Returns:
        Tuple[float, float, float, float]: t0, k, b, min_err
    """
    all_hits = np.array(points_xy)
    all_tdcs = np.array(tdc_times)
    track_hits_bool = np.ones(len(points_xy))  # 0 for bad hits, 1 for good hits

    best_result = None
    final_good_hits_ids = np.array([])
    final_bad_hits_ids = np.array([])
    min_err = math.inf
    bad_iters = 0

    for iter_id in range(max_iter):  # max_iteration: number of hits - 2
        logger.debug(f"track fitting iteration {iter_id}")

        # initialize
        good_hits_ids = np.flatnonzero(track_hits_bool == 1)
        good_hits = all_hits[good_hits_ids]
        good_tdcs = all_tdcs[good_hits_ids]
        bad_hits_ids = np.flatnonzero(track_hits_bool == 0)
        bad_hits = all_hits[bad_hits_ids]
        bad_tdcs = all_tdcs[bad_hits_ids]

        t0_init = sum(good_tdcs) / len(good_tdcs)
        t0_bound = (-50, max(good_tdcs))  # TODO: t0 could be < 0 or not?

        x = good_hits[:, 0]
        y = good_hits[:, 1]
        k_init, b_init, _, _, _ = scipy.stats.linregress(y, x)
        k_init_angle = math.atan(k_init)
        k_bound_down = math.tan(k_init_angle - k_angle_bound_range)
        k_bound_up = math.tan(k_init_angle + k_angle_bound_range)
        k_bound = (k_bound_down, k_bound_up)
        b_bound = (b_init - b_bound_range, b_init + b_bound_range)

        # function to be minimized for fitting
        def min_func(params):  # x = ky + b -> x - ky - b = 0
            t0 = params[0]
            k = params[1]
            b = params[2]
            result = 0
            for point, tdc_time in zip(good_hits, good_tdcs):
                radius = rt_function(tdc_time - t0)
                point_residual = get_point_err(
                    point, radius, k, b, negative_repel_factor=negative_repel_factor
                )
                point_err = rt_err_function(tdc_time - t0)
                result += (point_residual / point_err) ** 2
            return math.sqrt(result / len(good_hits))

        # fit track
        result = scipy.optimize.minimize(
            min_func,
            (t0_init, k_init, b_init),
            method=fit_method,
            bounds=[t0_bound, k_bound, b_bound],
        )
        current_err = min_func(result.x)
        logger.debug("min_func value:", current_err)
        t0 = result.x[0]
        k = result.x[1]
        b = result.x[2]

        # find the hit with largest error
        max_point_err_id = None
        max_point_err = -math.inf
        for idx in good_hits_ids:
            cur_hit = all_hits[idx]
            tdc_time = all_tdcs[idx]
            radius = rt_function(tdc_time - t0)
            point_err = get_point_err(cur_hit, radius, k, b)
            if point_err > max_point_err:
                max_point_err_id = idx
                max_point_err = point_err

        if current_err > min_err:
            if (
                bad_iters < iter_tolerance
            ):  # continue search even current result is worse
                bad_iters += 1
            else:
                break
        else:
            best_result = copy.deepcopy(result)
            final_good_hits_ids = good_hits_ids.copy()
            final_bad_hits_ids = bad_hits_ids.copy()
            min_err = current_err

        # remove or add back hits for fitting
        if max_point_err > off_track_hit_err:  # need further optimize
            track_hits_bool[max_point_err_id] = 0
        else:  # can't be further optimized
            # check if there is any good hit in bad_hits
            found_good_hits = False
            good_hit_in_bad_id = None
            min_point_err_in_bad = math.inf
            for idx in bad_hits_ids:
                cur_hit = all_hits[idx]
                tdc_time = all_tdcs[idx]
                radius = rt_function(tdc_time - t0)
                point_err = get_point_err(cur_hit, radius, k, b)
                if point_err < off_track_hit_err:
                    found_good_hits = True
                    if point_err < min_point_err_in_bad:
                        min_point_err_in_bad = point_err
                        good_hit_in_bad_id = idx
            if not found_good_hits:
                break
            else:
                track_hits_bool[good_hit_in_bad_id] = 1

    # prepare outputs
    t0 = best_result.x[0]
    k = best_result.x[1]
    b = best_result.x[2]
    logger.debug(f"track fitting results {result}")
    del track_hits_index[:]
    track_hits_index += final_good_hits_ids.tolist()
    for i in final_good_hits_ids:
        x = all_hits[i, 0]
        y = all_hits[i, 1]
        radius = rt_function(all_tdcs[i] - t0).item()
        track_hits.append((x, y, radius))
    for i in final_bad_hits_ids:
        x = all_hits[i, 0]
        y = all_hits[i, 1]
        radius = rt_function(all_tdcs[i] - t0).item()
        noise_hits.append((x, y, radius))

    return t0, k, b, min_err


def get_distance_point_line(point: Iterable[float], k: float, b: float) -> float:
    """Calculates point to line distance
    """
    x = point[0]
    y = point[1]
    return abs((x - k * y - b)) / math.sqrt(1 + k * k)


def get_point_err(
    point: Iterable[float],
    radius: float,
    k: float,
    b: float,
    negative_repel_factor: float = 1,
) -> float:
    """Evaluates how good is the hit on track
    """
    point_line_distance = get_distance_point_line(point, k, b)
    if point_line_distance >= radius:
        return point_line_distance - radius
    else:
        return math.sqrt(negative_repel_factor) * (radius - point_line_distance)
