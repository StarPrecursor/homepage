import logging
import pathlib
from typing import Any

import numpy as np
from smdt_rdv.analyze import track_search
from smdt_rdv.common.mapping import *
from smdt_rdv.common.smdt_const import *
from smdt_rdv.data_io import smdt_db
from smdt_rdv.job_manager import config_utils
from smdt_rdv.visualize import hits_display, spectrums

ANALYZER_DEFAULT_CFG = {
    "job": {
        "event_cuts": {"min_hits": 1, "max_hits": 1, "adc_cut": 0,},
        "track_cuts": {
            "min_hits_pass_event_cut": 0,
            "max_err": 7.5,
            "min_num_hits_on_track": 0,
            "max_num_hits_off_track": 500,
            "min_layers": 0,
        },
        "spectrums_cfg": {"include_unmatched_edges": False,},
    }
}

logger = logging.getLogger("smdt_rdv")


def analyze_single_file(db_path: str, file_id: int, config_dict: dict) -> dict:
    """ Analyzes single database file for spectrums, tracks...

    Args:
        db_path (str): input database file path
        file_id (int): file identification number
        config_dict (dict): job config dictionary

    Returns:
        dict: sMDT spectrums
    """
    # initialize:
    job_config = config_utils.sMDT_Job_Config_Section(ANALYZER_DEFAULT_CFG)
    job_config.update(config_dict)
    job_cfg = job_config.job
    path_cfg = job_config.path

    result_dir = pathlib.Path(path_cfg.results_out_dir)

    smdt_spectrums = {}
    for chamber_name in job_cfg.chamber_names:
        smdt_spectrums[chamber_name] = spectrums.init_spectrums_dict()

    # connect to database
    try:
        db_connect = smdt_db.connect(db_path)
        db_cursor = db_connect.cursor()
        logger.info(f"Successfully connected to database {db_path}")
    except:
        logger.error(f"Can't connect to database {db_path}")

    # collect events info from database
    for chamber_name in job_cfg.chamber_names:
        # prepare folders for plots
        if job_cfg.book_hits_plots:
            result_dir.joinpath(f"{chamber_name}/hits_display/file_{file_id}").mkdir(
                parents=True, exist_ok=True
            )
        if job_cfg.book_track_plots:
            result_dir.joinpath(f"{chamber_name}/track_display/file_{file_id}").mkdir(
                parents=True, exist_ok=True
            )
        # get events
        smdt_hits = smdt_db.get_events(db_cursor, min_hits=1, chamber_name=chamber_name)
        current_event_num = None
        tdc_lead_hits = []
        tdc_trail_hits = []
        # loop over hits
        for current_hit in smdt_hits:
            (rod_id, event_num, _, _, _, edge_type, _,) = current_hit
            # deal with new event
            if event_num != current_event_num:
                # process hits and save spectrums
                valid_hits = []
                process_hits(
                    job_cfg,
                    chamber_name,
                    tdc_lead_hits,
                    tdc_trail_hits,
                    valid_hits,
                    smdt_spectrums,
                )
                # plot hits
                if job_cfg.book_hits_plots:
                    min_hits_cut = job_cfg.event_cuts.min_hits
                    max_hits_cut = job_cfg.event_cuts.max_hits
                    if min_hits_cut < len(valid_hits) < max_hits_cut:
                        plot_title = f"Chamber: {chamber_name} MROD index: {rod_id} (RO side view)"
                        save_path = result_dir.joinpath(
                            f"{chamber_name}/hits_display/file_{file_id}/chamber_event_{event_num}_RO_hits_view.png"
                        )
                        hits_display.save_hits(
                            valid_hits, chamber_name, plot_title, save_path
                        )
                # try to search track if there are enough hits pass the event cuts
                if (
                    job_cfg.book_track_plots
                    and len(valid_hits)
                    >= job_cfg.track_cuts.min_hits_pass_event_cut
                ):
                    plot_title = (
                        f"Chamber: {chamber_name} MROD index: {rod_id} (RO side view)"
                    )
                    save_path = result_dir.joinpath(
                        f"{chamber_name}/track_display/file_{file_id}/chamber_event_{event_num}_RO_hits_view.png"
                    )
                    search_tracks(
                        job_cfg,
                        chamber_name,
                        valid_hits,
                        smdt_spectrums,
                        plot_title,
                        save_path,
                    )

                current_event_num = event_num
                if job_cfg.max_events_per_file and job_cfg.max_events_per_file > 0:
                    if event_num > job_cfg.max_events_per_file:
                        break
            # get leading tdc
            if edge_type == 1:
                tdc_lead_hits.append(current_hit)
            # get trailing tdc
            elif edge_type == 0:
                tdc_trail_hits.append(current_hit)
            # unknown case
            else:
                logger.error(f"Unknown TDC edge type: {edge_type}")

    return smdt_spectrums


def get_dict_value_with_default(key: str, source_dict: dict, default_dict: dict) -> Any:
    if key in source_dict:
        if source_dict[key] is not None:  # not None
            return source_dict[key]
    if key in default_dict:
        return default_dict[key]
    else:
        logger.warning(f"Undefined value used for {key}")
        return None


def process_hits(
    job_cfg,
    chamber_name: str,
    tdc_lead_hits: list,
    tdc_trail_hits: list,
    valid_hits: list,
    smdt_spectrums: dict,
) -> None:
    """Processes hits and make spectrums

    Note:
        input tdc_lead/trail_hits will be emptied
        valid_hits will be filled
        input smdt_spectrums with be changed

    Args:
        job_cfg: confit contents in - job - section
        chamber_name (str): name of the chamber to be studied
        tdc_lead_hits (list): collection of leading TDC edge hits
        tdc_trail_hits (list): collection of trailing TDC edge hits
        valid_hits (list): output hits that pass the selection
        smdt_spectrums (dict): output smdt_spectrums to be updated
    """
    # save spectrums
    if tdc_lead_hits:
        hits_record = (
            []  # element: (mezz, channel)
        )  # used to avoid multiple hits in one place
        while tdc_lead_hits:
            (_, _, _, mezz, channel, _, tdc_time,) = tdc_lead_hits.pop()
            # add to leading TDC spectrum
            logger.debug(f"Add TDC entry to mezz {mezz}, channel {channel}")
            # find matched trailing TDC
            trail_match = None
            for (i, (_, _, _, mezz_, channel_, _, tdc_time_),) in enumerate(
                tdc_trail_hits
            ):
                if mezz_ == mezz and channel_ == channel and tdc_time_ > tdc_time:
                    trail_match = i

                    adc_time = tdc_time_ - tdc_time
                    if (adc_time > job_cfg.event_cuts.adc_cut) and (
                        (mezz, channel) not in hits_record
                    ):
                        if job_cfg.book_spectrums:
                            current_spectrums = smdt_spectrums[chamber_name][mezz][
                                channel
                            ]
                            current_spectrums["tdc_lead"][int(tdc_time)] += 1
                            # add trailing TDC entry
                            current_spectrums["tdc_trail"][int(tdc_time_)] += 1
                            # add adc entry
                            current_spectrums["adc"][int(adc_time)] += 1
                        valid_hits.append(
                            (mezz, channel, tdc_time, tdc_time_, adc_time,)
                        )
                        hits_record.append((mezz, channel))
                    break
            # remove matched trailing edge
            if trail_match:
                tdc_trail_hits.pop(trail_match)
        # add remaining trailing TDC entries if any
        if job_cfg.book_spectrums and job_cfg.spectrums_cfg.include_unmatched_edges:
            for _, _, _, mezz_, channel_, _, tdc_time_ in tdc_lead_hits:
                current_spectrums = smdt_spectrums[chamber_name][mezz][channel]
                current_spectrums["tdc_lead"][int(tdc_time_)] += 1
            for _, _, _, mezz_, channel_, _, tdc_time_ in tdc_trail_hits:
                current_spectrums = smdt_spectrums[chamber_name][mezz][channel]
                current_spectrums["tdc_trail"][int(tdc_time_)] += 1
    # empty hits
    del tdc_lead_hits[:]
    del tdc_trail_hits[:]


def search_tracks(
    job_cfg,
    chamber_name: str,
    valid_hits: list,
    smdt_spectrums: dict,
    plot_title: str,
    save_path: pathlib.Path,
):
    if (
        job_cfg.book_track_plots
        and len(valid_hits) >= job_cfg.track_cuts.min_hits_pass_event_cut
    ):
        fig, ax = hits_display.create_smdt_tubes_axes(title=plot_title)
        hits_display.plot_smdt_tubes(ax, chamber_name)
        hits_positions = []
        hits_tdc_ns = []
        for (mezz, channel, tdc_time, _, _) in valid_hits:
            # plot hit based on leading TDC
            hit_layer, hit_column = mezz_to_tube_layer_column(
                chamber_name, mezz, channel
            )
            hit_xy = mezz_to_tube_x_y(chamber_name, mezz, channel)
            hits_positions.append(hit_xy)
            hits_tdc_ns.append(tdc_time * 25 / 32)  # convert digit to ns

        # fitting
        track_hits_index = []
        track_hits = []
        noise_hits = []
        t0_ns, k, b, min_err = track_search.fit_track(
            hits_positions,
            hits_tdc_ns,
            track_hits_index=track_hits_index,
            track_hits=track_hits,
            noise_hits=noise_hits,
            fit_method=job_cfg.track_fit.fit_method,
            negative_repel_factor=job_cfg.track_fit.negative_repel_factor,
            off_track_hit_err=job_cfg.track_fit.off_track_hit_err,
            k_angle_bound_range=job_cfg.track_fit.k_angle_bound_range,
            b_bound_range=job_cfg.track_fit.b_bound_range,
            max_iter=job_cfg.track_fit.max_iter,
            iter_tolerance=job_cfg.track_fit.iter_tolerance,
        )
        y_set = set()
        for hit in track_hits:
            x, y, r = hit
            y_set.add(y)
        num_layers = len(y_set)

        # select tracks to keep
        num_hits_on_track = len(track_hits)
        num_hits_off_track = len(noise_hits)
        should_save_track = True
        if (
            job_cfg.track_cuts.max_err
            and min_err > job_cfg.track_cuts.max_err
        ):
            should_save_track = False
        if num_hits_on_track < job_cfg.track_cuts.min_num_hits_on_track:
            should_save_track = False
        if num_hits_off_track > job_cfg.track_cuts.max_num_hits_off_track:
            should_save_track = False
        if num_layers < job_cfg.track_cuts.min_layers:
            should_save_track = False
        if should_save_track:
            # plot track and hits
            for hit in track_hits:
                x, y, r = hit
                y_set.add(y)
                hits_display.plot_hit_x_y(ax, x, y, r)
            for hit in noise_hits:
                x, y, r = hit
                hits_display.plot_hit_x_y(
                    ax, x, y, r, fill_color="brown", mark_color="brown"
                )
            y = np.linspace(0, 150, 50)
            ax.plot(k * y + b, y, color="red", linewidth=0.5)
            fig.savefig(save_path)
        hits_display.close_figures()

        # plot spectrums on track
        if should_save_track and job_cfg.book_spectrums:
            for index in track_hits_index:
                mezz, channel, tdc_time, tdc_time_, adc_time = valid_hits[index]
                current_spectrums = smdt_spectrums[chamber_name][mezz][channel]
                current_spectrums["tdc_lead_track"][int(tdc_time)] += 1
                # add trailing TDC entry
                current_spectrums["tdc_trail_track"][int(tdc_time_)] += 1
                # add TDC - T0 entry
                t0 = t0_ns / (25 / 32)
                current_spectrums["tdc_t0_track"][int(tdc_time - t0)] += 1
                # add adc entry
                current_spectrums["adc_track"][int(adc_time)] += 1

                mezz, channel, tdc_time, tdc_time_, adc_time = valid_hits[index]
                current_spectrums = smdt_spectrums[chamber_name][mezz][channel]
                current_spectrums["tdc_lead_track"][int(tdc_time)] += 1
                # add trailing TDC entry
                current_spectrums["tdc_trail_track"][int(tdc_time_)] += 1
                # add TDC - T0 entry
                t0 = t0_ns / (25 / 32)
                current_spectrums["tdc_t0_track"][int(tdc_time - t0)] += 1
                # add adc entry
                current_spectrums["adc_track"][int(adc_time)] += 1
